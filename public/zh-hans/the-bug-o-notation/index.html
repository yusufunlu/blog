function FormStatus() {

    step: 'initial'
  });

  function handleSubmit(e) {

    if (state.step === 'pending') {
      // Don't allow to submit twice
      return;

    setState({ step: 'pending' });
    submitForm().then(() => {
      setState({ step: 'success' });

      setState({ step: 'error', error });
    });

      content = <Spinner />;

      content = <SuccessMessage />;

  }

  return (
    <form onSubmit={handleSubmit}>
      {content}
    </form>
  );
}
